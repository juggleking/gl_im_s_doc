/* Name     : ery.prco */
/* What     : scheme to derive bloodgroup from erysubtests */
/* Basetable: action */
/*  BGEryA   BGEryB BGEryAB  BGERY */
/* ------------------------------- */
/*    +        -         +     A   */
/*    -        +         +     B   */
/*    +        +         +     AB  */
/*    -        -         -     O   */ 
string bgery;
 if BGEryA = "+" and
    BGEryB = "-" and
    BGEryAB = "+" then
         Message(" bloodgroup A");
      bgery := "A"; 
 endif;
 if BGEryA = "-" and
    BGEryB = "+" and
    BGEryAB = "+" then
         Message(" bloodgroup B");
      bgery := "B"; 
 endif;
 if BGEryA = "+" and
    BGEryB = "+" and
    BGEryAB = "+" then
         Message(" bloodgroup AB");
      bgery := "AB"; 
 endif;
 if BGEryA = "-" and
    BGEryB = "-" and
    BGEryAB = "-" then
         Message(" bloodgroup O");
      bgery := "O"; 
 endif;
  if bgery <> "A" and 
     bgery <> "B" and
     bgery <> "O" and
     bgery <> "AB" then
   message("Invalid test combination");  
endif;
return(bgery);
